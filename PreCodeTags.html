                        enum enMathOparations {Add, Sub, Mult, Div};
        
                        int SumOfTwoNumbers(int Num1, int Num2 , enMathOparations MathOprastion)
                        {
                           switch(MathOprastion)
                           {
                             case enMathOparations.Add :
                                    return Num1 + Num2;
                             break;
        
                            case enMathOparations.Sub :
                            return Num1 - Num2;
                            break;
        
                            case enMathOparations.Mult :
                            return Num1 * Num2;
                            break;
        
                            case enMathOparations.Div :
                            return Num1 / Num2;
                            break;
        
                           }
                        }
